Alert

Toast flash notifications — fixed overlay, auto-dismiss, zero layout impact. Alerts shown on page load have already dismissed; use buttons to re-trigger.

{{ c.alert_info | safe }} {{ c.alert_success | safe }} {{ c.alert_warning | safe }} {{ c.alert_error | safe }}
Show MCP usage 
add_component(
    component_type="alert",
    config={
        "message": "Operation completed successfully!",
        "type": "success",      # info | success | warning | error
        "duration": 4000,       # ms before auto-dismiss (0 = persistent)
        "position": "top-right", # top-right | top-center | top-left | bottom-*
        "dismissable": True     # show × close button
    }
)
